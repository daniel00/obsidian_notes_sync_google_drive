## 오름차순 (reference code 거의 그대로)

```cpp
#include <iostream>
static const int MAX_NUM = 100;

void quickSort(int input[MAX_NUM], int first, int last)
{
    int pivot;
    int i;
    int j;
    int temp;

    if (first < last)
    {
        pivot = first;
        i = first;
        j = last;

        while (i < j)
        {
            while (input[i] <= input[pivot] && i < last)
            {
                i++;
            }
            while (input[j] > input[pivot])
            {
                j--;
            }
            if (i < j)
            {
                temp = input[i];
                input[i] = input[j];
                input[j] = temp;
            }
        }

        temp = input[pivot];
        input[pivot] = input[j];
        input[j] = temp;

        quickSort(input, first, j - 1);
        quickSort(input, j + 1, last);
    }
}

int main() {

    int arr[6] = { 4, 2, 5, 1 };

    quickSort(arr, 0, 3);

    for (int num : arr) {
        std::cout << num << '\n';
    }

    return 0;
}
```

## Custom Sort

(y값이 작은것 우선 / x값이 작은 것)

```cpp
#include <iostream>
static const int MAX_NUM = 100;

struct Node {
    int y, x;
};

bool compare(Node t1, Node t2) {
    //t1이 나열했을때 더 왼쪽에 배치되어야하면 true 리턴
    if (t1.y != t2.y) return t1.y < t2.y;
	return t1.x < t2.x;
}

void quickSort(Node input[MAX_NUM], int first, int last)
{
    int pivot;
    int i;
    int j;
    Node temp;

    if (first < last)
    {
        pivot = first;
        i = first;
        j = last;

        while (i < j)
        {
            //컴페어
            while (compare(input[i], input[pivot]) && i < last)
            {
                i++;
            }
            while (!compare(input[j], input[pivot])) //컴페어
            {
                j--;
            }
            if (i < j)
            {
                temp = input[i];
                input[i] = input[j];
                input[j] = temp;
            }
        }

        temp = input[pivot];
        input[pivot] = input[j];
        input[j] = temp;

        quickSort(input, first, j - 1);
        quickSort(input, j + 1, last);
    }
}

int main() {

    Node arr[4] = { {1, 2}, {6, 4}, {5, 5}, {5, 1} };

    quickSort(arr, 0, 3);

    for (Node num : arr) {
        std::cout << num.y << "," << num.x << '\n';
    }

    return 0;
}
```
