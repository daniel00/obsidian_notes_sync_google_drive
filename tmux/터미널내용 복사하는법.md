#### tmux에서 터미널 내용 복사하는 법 정리
리눅스에서는 xclip를 설치하고, sudo apt install xclip
다음처럼 설정하면 된다.
ctrl + b + [ 
로 복사모드에 진입

v, V 로 선택 . space도 동작. 
y로 복사.

tmux창에 붙여넣기는, p 또는 ctrl b ]
다른 앱에 붙여 넣기는 ctrl v


```bash
bind-key -Tcopy-mode-vi 'v' send -X begin-selection
bind-key -Tcopy-mode-vi 'y' send -X copy-pipe-and-cancel 'xclip -in -selection clipboard'
```


전체 파일 내용
```bash
set -sg escape-time 10

bind h select-pane -L
bind j select-pane -D
bind k select-pane -U
bind l select-pane -R

# ~/.tmux.conf 에 추가
bind = select-layout even-horizontal
bind - select-layout even-vertical


set-option -g mouse on


#plugin
set -g @plugin 'tmux-plugins/tmux-yank'

set -g @yank_selection 'clipboard'



# vi 모드 사용
setw -g mode-keys vi
set -g status-keys vi

# vim 스타일 복붙 모드
# unbind [
# bind Escape copy-mode
# unbind p
# bind p paste-buffer

bind-key -Tcopy-mode-vi 'v' send -X begin-selection
bind-key -Tcopy-mode-vi 'y' send -X copy-pipe-and-cancel 'xclip -in -selection clipboard'



# pbcopy 대신 사용자의 터미널(맥북)로 클립보드 데이터를 전달 (OSC 52 이용)
# bind -T copy-mode-vi y send -X copy-pipe-and-cancel "set-clipboard"

set -s set-clipboard on

# 마우스 드래그 종료 시 시스템 클립보드로 복사 (vi 모드 기준)
bind-key -T copy-mode-vi MouseDragEnd1Pane send-keys -X copy-pipe-and-cancel "xclip -se c -i"

# set -g status-bg colour124 # 어두운 빨간색
# set -g status-bg colour51 # 밝은 시안
set -g status-bg colour14 # 시안

# set -g status-fg white
set -g status-fg black



# 1. 터미널의 기본 지우기 신호를 명시적으로 지정
set -g default-terminal "xterm-256color"


```
