##### ST LINK V2를 리눅스에서 인식 시키는 법

###### rules 파일 생성
sudo vim /etc/udev/rules.d/49-stlinkv2.rules

###### 아래 내용 입력 & 저장

```text
# ST-LINK/V2
SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="3748", \
    MODE:="0666", \
    SYMLINK+="stlinkv2_%n"

# ST-LINK/V2-1
SUBSYSTEMS=="usb", ATTRS{idVendor}=="0483", ATTRS{idProduct}=="374b", \
    MODE:="0666", \
    SYMLINK+="stlinkv2-1_%n"
```


###### 설정 적용하기
 
 ```bash
sudo udevadm control --reload-rules
sudo udevadm trigger
 ```

테스트용으로 LED를 깜박이는 코드를 다음처럼 작성하고 make flash로 실행한다.

LED는 PC13에 연결 되어 있다.
###### main.c

```c
#include <stdint.h>

// 스택 끝 주소 (20KB RAM 기준: 0x20000000 + 20*1024)
extern void _estack(void);
void Reset_Handler(void);
int main(void);

// 벡터 테이블 정의 (플래시 가장 앞부분에 배치되어야 함)
__attribute__((section(".vectors")))
const void* vector_table[] = {
    (void*)0x20005000,  // 초기 스택 포인터
    (void*)Reset_Handler // 리셋 핸들러
};

void Reset_Handler(void) {
    main();
    while(1);
}

#define RCC_APB2ENR  (*(volatile uint32_t *)0x40021018)
#define GPIOC_CRH    (*(volatile uint32_t *)0x40011004)
#define GPIOC_ODR    (*(volatile uint32_t *)0x4001100C)

void delay(volatile uint32_t count) {
    while (count--) __asm__("nop");
}

int main(void) {
    RCC_APB2ENR |= (1 << 4);      // Port C 클럭 활성화
    GPIOC_CRH &= ~(0xF << 20);    // PC13 설정 초기화
    GPIOC_CRH |= (0x2 << 20);     // PC13 출력 모드 설정

    while (1) {
        GPIOC_ODR ^= (1 << 13);   // LED 토글
        delay(500000);
    }
}
```


###### startup.s

```asm
.syntax unified
.cpu cortex-m3
.thumb

.word _estack            /* 스택 시작 주소 */
.word Reset_Handler      /* 리셋 핸들러 주소 */

.section .text.Reset_Handler
.type Reset_Handler, %function

Reset_Handler:
    ldr sp, =_estack     /* 스택 포인터 초기화 */
    bl main              /* main 함수로 분기 */
loop:
    b loop               /* 혹시 main이 종료되면 무한 루프 */
```


Makefile

```Makefile
# 프로젝트 이름
TARGET = main

# 툴체인 설정
CC      = arm-none-eabi-gcc
AS      = arm-none-eabi-as
OBJCOPY = arm-none-eabi-objcopy

# MCU 사양 및 컴파일 플래그
# -nostartfiles: 표준 스타트업 파일을 사용하지 않음 (직접 만든 startup.s 사용)
CFLAGS  = -mcpu=cortex-m3 -mthumb -O0 -g -nostartfiles
ASFLAGS = -mcpu=cortex-m3 -mthumb

# 소스 파일 및 오브젝트 파일
SRCS = main.c
ASMS = startup.s
# 링크 순서가 중요하므로 startup.o를 가장 앞에 배치합니다.
OBJS = startup.o main.o

# 기본 타겟
all: $(TARGET).bin

# C 컴파일
%.o: %.c
	$(CC) -c $(CFLAGS) $< -o $@

# 어셈블리 컴파일
%.o: %.s
	$(AS) $(ASFLAGS) $< -o $@

# 링크 과정
# -Ttext 0x08000000: 코드가 시작될 주소
# --defsym=_estack: 스택 포인터 위치 정의 (20KB RAM 끝: 0x20005000)
$(TARGET).elf: $(OBJS)
	arm-none-eabi-ld -Ttext 0x08000000 --defsym=_estack=0x20005000 $(OBJS) -o $@

# 바이너리 추출
$(TARGET).bin: $(TARGET).elf
	$(OBJCOPY) -O binary $< $@

# 플래싱 (st-flash 사용)
flash: $(TARGET).bin
	st-flash --reset write $(TARGET).bin 0x08000000

# 정리
clean:
	rm -f *.o *.elf *.bin

.PHONY: all flash clean
```
