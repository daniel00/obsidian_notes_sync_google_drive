## 오름차순 (reference code 거의 그대로)

```cpp
#include <iostream>
static const int MAX_NUM = 100;

void quickSort(int input[MAX_NUM], int first, int last)
{
    int pivot;
    int i;
    int j;
    int temp;

    if (first < last)
    {
        pivot = first;
        i = first;
        j = last;

        while (i < j)
        {
            while (input[i] <= input[pivot] && i < last)
            {
                i++;
            }
            while (input[j] > input[pivot])
            {
                j--;
            }
            if (i < j)
            {
                temp = input[i];
                input[i] = input[j];
                input[j] = temp;
            }
        }

        temp = input[pivot];
        input[pivot] = input[j];
        input[j] = temp;

        quickSort(input, first, j - 1);
        quickSort(input, j + 1, last);
    }
}

int main() {

    int arr[6] = { 4, 2, 5, 1 };

    quickSort(arr, 0, 3);

    for (int num : arr) {
        std::cout << num << '\n';
    }

    return 0;
}
```

## Custom Sort
아래는 정상 동작을 확인 한 코드이다.
compare() 함수의 return t1.y < t2.y, return t1.x < t2.x 에 등호가 들어가면 안된다는 점에 유의해야 한다.
 "strict weak ordering" 으로 검색해 보자.
```c++
#include <iostream>
#include <stdio.h>

static const int MAX_NUM = 100000;

struct Node
{
    int y;
    int x;
};

// Node arr[MAX_NUM] = {0};

bool compare(Node t1, Node t2)
{
    if (t1.y != t2.y)
        return t1.y > t2.y;  //y먼저 내림차순으로 정렬
	else
	    return t1.x < t2.x;   //y가 같을 경우, x를 오름 차순으로 정렬
}

void quickSort(Node input[MAX_NUM], int first, int last)
{
    int pivot;
    int i;
    int j;
    Node temp;

    if (first < last)
    {
        pivot = first;
        i = first;
        j = last;

        while (i < j)
        {
            // 컴페어
            while (i < last && compare(input[i], input[pivot]))  //경계 조건에 유의
            {
                i++;
            }
            while (j > first && !compare(input[j], input[pivot])) // 컴페어
            {
                j--;
            }

            if (i < j)
            {
                temp = input[i];
                input[i] = input[j];
                input[j] = temp;
            }
        }

        temp = input[pivot];
        input[pivot] = input[j];
        input[j] = temp;

        quickSort(input, first, j - 1);
        quickSort(input, j + 1, last);
    }
}

int main()
{

    Node arr[4] = {{1, 2}, {6, 4}, {5, 5}, {5, 1}};

    quickSort(arr, 0, 3);

    for (Node num : arr)
    {
        std::cout << num.y << "," << num.x << '\n';
    }

    return 0;
}


```
